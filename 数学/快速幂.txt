public FastPower {
	public int fastPower(int a, int b) {// a^b
		int result = 1;
		while(b > 0) {
			int i = 1;
			int temp = a;
			do {
				i = i * 2;
				if(i <= b)
					temp = temp * temp;	
			} while(i < b);
			
			if(i == b)
				b = b - i;
			else 
				b = b - i / 2;
			result = result * temp;
		}
		return result;
	}
}
